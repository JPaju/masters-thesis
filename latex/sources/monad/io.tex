\begin{minted}{scala}
case class IO[A](run: () => A)

given Monad[IO] = new:
  def pure[A](a: A): IO[A] = IO(() => a)
  extension [A](io: IO[A])
    def flatMap[B](f: A => IO[B]): IO[B] =
      IO(() => f(io.run()).run())
\end{minted}
