\begin{minted}{scala}
def optionToEither[A]: Option[A] => Either[Unit, A] = {
  case None    => Left(())
  case Some(a) => Right(a)
}

def eitherToOption[A]: Either[Unit, A] => Option[A] = {
  case Left(()) => None
  case Right(a) => Some(a)
}
\end{minted}
