\begin{minted}{ocaml}
structural ability Exception e where
  raise : e -> a

toOptional : '{Exception e} a -> Optional a
toOptional mightThrow =
  handle !mightThrow with cases
    { raise e -> c }  -> None
    { a }             -> Some a
